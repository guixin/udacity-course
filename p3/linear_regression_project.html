def matxMultiply(a, b):
    if len(a[0]) != len(b):
        raise ValueError("a column size is {} and b row size is {} ".format(len(a[0]), len(b)))

    result = []
    for i in range(len(a)):
        vet = []
        for j in range(len(b[0])):
            v = 0
            for k in range(len(b)):
                v += a[i][k] * b[k][j]
            vet.append(v)
        result.append(vet)

    return result

""" Gaussian Jordan  Ax = b.
    
        A:  
        b: 
        decPts: 4
        epsilon: 0 1.0e-16
        
     x  Ax = b 
    None Ab 
    None A 
"""

def gj_Solve(A, b, decPts=4, epsilon=1.0e-16):
    if len(A) != len(b):
        return None
    Ab = augmentMatrix(A, b)
    row, line = len(Ab[0]), len(Ab)

    for i in range(row - 1):
        max_val = abs(Ab[i][i])
        max_val_index = i
        for j in range(i + 1, line):
            tmp = max(abs(Ab[j][i]), max_val)
            if tmp > max_val:
                max_val = tmp
                max_val_index = j
        if abs(max_val) < epsilon:
            return None
        if max_val_index != line:
            Ab[i], Ab[max_val_index] = Ab[max_val_index], Ab[i]
        denominator = float(Ab[i][i])
        for index in range(i, row):
            Ab[i][index] *= 1 / denominator
        for j in range(line):
            if j != i and Ab[j][i] != 0:
                addScaledRow(Ab, j, i, -Ab[j][i])
    result = [[round(item[-1], decPts)]for item in Ab]
    return result

def calculateMSE(X, Y, m, b):
    if not X and not Y:
        raise ValueError

    x_size = len(X)
    y_size = len(Y)
    n = max(x_size, y_size)

    v = 0
    for i in range(n):
        x, y = 0, 0
        if i < x_size:
            x = X[i]
        if i < y_size:
            y = Y[i]
        v += (y - m * x - b) ** 2
    if n == 0:
        return v
    return v / float(n)
